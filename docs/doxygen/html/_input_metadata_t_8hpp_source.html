74 if constexpr (FundamentalType<T>) {

102template <typename T>
103struct StringSerializationHelper {
104 static auto Serialize(void const* objPtr, SP::SlidingBuffer& bytes) -> void {
105 std::string_view str;
106 if constexpr (StringLiteral<T>) {
107 str = static_cast<const char*>(objPtr);
108 } else if constexpr (StringType<T>) {
109 str = *static_cast<T const*>(objPtr);
110 }
111 uint32_t size = str.size();
112 bytes.append(reinterpret_cast<const uint8_t*>(&size), sizeof(size));
113 bytes.append(reinterpret_cast<const uint8_t*>(str.data()), size);
114 }
115
116 static auto Deserialize(void* objPtr, SP::SlidingBuffer const& bytes) -> void {
117 if (bytes.size() < sizeof(uint32_t)) {
118 throw std::runtime_error("Buffer too small for string size");
119 }
120
121 uint32_t size;
122 std::memcpy(&size, bytes.data(), sizeof(size));
123
124 if (bytes.size() < sizeof(size) + size) {
125 throw std::runtime_error("Buffer too small for string data");
126 }
127
128 if constexpr (std::is_same_v<T, std::string>) {
129 auto& str = *static_cast<std::string*>(objPtr);
130 str.assign(reinterpret_cast<const char*>(bytes.data() + sizeof(size)), size);
131 } else {
132 throw std::runtime_error("Can only deserialize to std::string");
133 }
134 }
135
136 static auto DeserializePop(void* objPtr, SP::SlidingBuffer& bytes) -> void {
137 StringSerializationHelper<T>::Deserialize(objPtr, bytes);
138 bytes.advance(sizeof(uint32_t) + *reinterpret_cast<const uint32_t*>(bytes.data()));
139 }
140};
