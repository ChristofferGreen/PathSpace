205 static CGEventRef MouseTapCallback(CGEventTapProxy /*proxy*/, CGEventType type, CGEventRef event, void* refcon) {
206 auto self = static_cast<PathIOMouse*>(refcon);
207 if (!self) return event;
208
209 // Re-enable tap if it gets disabled
210 if (type == kCGEventTapDisabledByTimeout || type == kCGEventTapDisabledByUserInput) {

212 return event;
213 }
214
215 // Map CG events to high-level mouse events
216 switch (type) {
217 case kCGEventMouseMoved:
218 case kCGEventLeftMouseDragged:
219 case kCGEventRightMouseDragged:
220 case kCGEventOtherMouseDragged: {
221 Event ev;
222 ev.type = MouseEventType::Move;
223 ev.dx = static_cast<int>(CGEventGetIntegerValueField(event, kCGMouseEventDeltaX));
224 ev.dy = static_cast<int>(CGEventGetIntegerValueField(event, kCGMouseEventDeltaY));
225 ev.timestampNs = CGEventGetTimestamp(event);
226 {
227 std::lock_guard<std::mutex> lg(self->mutex_);
228 self->queue_.push_back(ev);
229 }
230 self->cv_.notify_all();
231 break;
232 }
233 case kCGEventLeftMouseDown:
234 case kCGEventRightMouseDown:
235 case kCGEventOtherMouseDown: {
236 Event ev;
237 ev.type = MouseEventType::ButtonDown;
238 ev.button = (type == kCGEventLeftMouseDown) ? MouseButton::Left
239 : (type == kCGEventRightMouseDown) ? MouseButton::Right
240 : MouseButton::Middle;
241 ev.timestampNs = CGEventGetTimestamp(event);
242 {
243 std::lock_guard<std::mutex> lg(self->mutex_);
244 self->queue_.push_back(ev);
245 }
246 self->cv_.notify_all();
247 break;
248 }
249 case kCGEventLeftMouseUp:
250 case kCGEventRightMouseUp:
251 case kCGEventOtherMouseUp: {
252 Event ev;
253 ev.type = MouseEventType::ButtonUp;
254 ev.button = (type == kCGEventLeftMouseUp) ? MouseButton::Left
255 : (type == kCGEventRightMouseUp) ? MouseButton::Right
256 : MouseButton::Middle;
257 ev.timestampNs = CGEventGetTimestamp(event);
258 {
259 std::lock_guard<std::mutex> lg(self->mutex_);
260 self->queue_.push_back(ev);

263 break;

286 void osInit_() {
287#if defined(PATHIO_BACKEND_MACOS)
288 // Only initialize OS hooks when explicitly enabled (backend = OS)
289 if (mode_ != BackendMode::OS) {
290 fallbackActive_.store(false, std::memory_order_release);
291 osReady_.store(false, std::memory_order_release);
292 return;
293 }
294
295 CFRunLoopRef current = CFRunLoopGetCurrent();
296 if (runLoopRef_ && runLoopRef_ != current) {
297 osShutdown_();
298 }
299 if (eventTap_) {
300 // Already initialized for this runloop
301 return;
302 }
303
304 CGEventMask mask = CGEventMaskBit(kCGEventMouseMoved) | CGEventMaskBit(kCGEventLeftMouseDragged) |
305 CGEventMaskBit(kCGEventRightMouseDragged) | CGEventMaskBit(kCGEventOtherMouseDragged) |
306 CGEventMaskBit(kCGEventLeftMouseDown) | CGEventMaskBit(kCGEventLeftMouseUp) |
307 CGEventMaskBit(kCGEventRightMouseDown) | CGEventMaskBit(kCGEventRightMouseUp) |
308 CGEventMaskBit(kCGEventOtherMouseDown) | CGEventMaskBit(kCGEventOtherMouseUp) |
309 CGEventMaskBit(kCGEventScrollWheel);
310
311 eventTap_ = CGEventTapCreate(kCGHIDEventTap,
312 kCGHeadInsertEventTap,
313 kCGEventTapOptionDefault,
314 mask,
315 &MouseTapCallback,
316 this);
317 if (!eventTap_) {
318 // Could not create tap (permissions?). Fall back to no OS sourcing; tests still function via insert()
319 fallbackActive_.store(false, std::memory_order_release);
320 return;
321 } else {
322 fallbackActive_.store(false, std::memory_order_release);
323 }
324
325 eventSrc_ = CFMachPortCreateRunLoopSource(kCFAllocatorDefault, eventTap_, 0);
326 runLoopRef_ = current;
327
328 if (eventSrc_ && runLoopRef_) {
329 CFRunLoopAddSource(runLoopRef_, eventSrc_, kCFRunLoopCommonModes);
330 CGEventTapEnable(eventTap_, true);
331 osReady_.store(true, std::memory_order_release);
332 }
333#else
334 (void)path; // no-op on non-macOS builds
335#endif

361 // Poll once: service the current thread's runloop briefly
362 void osPollOnce_() {
363#if defined(PATHIO_BACKEND_MACOS)
364 if (!osReady_.load(std::memory_order_acquire)) {
365 osInit_();
366 }
367 if (mode_ != BackendMode::OS) {
368 // Do nothing when not in OS mode; tests and apps can use insert("/events") to feed data
369 std::this_thread::sleep_for(std::chrono::milliseconds(10));
370 return;
371 }
372 if (fallbackActive_.load(std::memory_order_acquire)) {
373 // Permission-less fallback disabled by default
374 std::this_thread::sleep_for(std::chrono::milliseconds(10));
375 return;
376 }
377 if (runLoopRef_) {
378 // Process pending events for a short slice without blocking indefinitely
379 CFRunLoopRunInMode(kCFRunLoopDefaultMode, 0.02, true);
380 } else {
381 std::this_thread::sleep_for(std::chrono::milliseconds(10));
382 }
383#else
384 // Non-macOS / backend disabled: no-op
385 std::this_thread::sleep_for(std::chrono::milliseconds(10));
386#endif
387 }
