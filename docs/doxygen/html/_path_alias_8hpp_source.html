114 // Join targetPrefix_ and a tail path, ensuring exactly one slash at the boundary.
115 static std::string joinPaths_(std::string const& prefix, std::string const& tail) {
116 if (prefix.empty())
117 return tail;
118 if (tail.empty())
119 return prefix;
120
121 bool prefixEndsWithSlash = !prefix.empty() && prefix.back() == '/';
122 bool tailStartsWithSlash = !tail.empty() && tail.front() == '/';
123
124 if (prefixEndsWithSlash && tailStartsWithSlash) {
125 return prefix + tail.substr(1);
126 } else if (!prefixEndsWithSlash && !tailStartsWithSlash) {
127 return prefix + "/" + tail;
128 } else {

130 }
131 }
