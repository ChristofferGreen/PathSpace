161 // ---- PathSpaceBase override: typed read/take ----
162
163 // Serve typed PointerEvent with peek/pop and optional blocking semantics.
164 // - If options.doPop is true: pop the front event into obj; otherwise peek without consuming.
165 // - If the queue is empty:
166 // * If options.doBlock is false: return NoObjectFound.
167 // * If options.doBlock is true: wait until timeout for an event to arrive; return Timeout on expiry.
168 auto out(Iterator const& /*path*/,
169 InputMetadata const& inputMetadata,
170 Out const& options,
171 void* obj) -> std::optional<Error> override {
172 // Type-check: only support Event payloads here
173 if (inputMetadata.typeInfo != &typeid(Event)) {
174 return Error{Error::Code::InvalidType, "PathIOPointerMixer only supports PointerEvent"};
175 }
176 if (obj == nullptr) {
177 return Error{Error::Code::MalformedInput, "Null output pointer for PathIOPointerMixer::out"};
178 }
179
180 // Fast path: try without blocking
181 {
182 std::lock_guard<std::mutex> lg(mutex_);
183 if (!queue_.empty()) {
184 Event ev = queue_.front();
185 if (options.doPop) {
186 queue_.pop_front();
187 }
188 *reinterpret_cast<Event*>(obj) = ev;
189 return std::nullopt;
190 }
191 }
192
193 // No event and non-blocking read requested
194 if (!options.doBlock) {
195 return Error{Error::Code::NoObjectFound, "No pointer event available"};
196 }
197
198 // Blocking path: wait until an event is available or timeout expires
199 auto deadline = std::chrono::steady_clock::now() +
200 std::chrono::duration_cast<std::chrono::steady_clock::duration>(options.timeout);
201
202 {
203 std::unique_lock<std::mutex> lk(mutex_);
204 // Wait predicate: queue is non-empty or deadline reached
205 if (!cv_.wait_until(lk, deadline, [this] { return !queue_.empty(); })) {
206 return Error{Error::Code::Timeout, "Timed out waiting for pointer event"};
207 }
208 // Event is available now
209 if (queue_.empty()) {
210 // Rare race: event consumed by another thread before we reacquired the lock
211 return Error{Error::Code::NoObjectFound, "No pointer event available after wake"};
212 }
213 Event ev = queue_.front();
214 if (options.doPop) {
215 queue_.pop_front();
216 }
217 *reinterpret_cast<Event*>(obj) = ev;
218 }
219 return std::nullopt;
220 }
