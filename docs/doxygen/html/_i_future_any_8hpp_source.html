69 std::scoped_lock<std::mutex> lg(mutex_);
70 return ready_;

88 if (dest == nullptr) return false;
89 std::scoped_lock<std::mutex> lg(mutex_);
90 if (!ready_ || !value_.has_value()) return false;
91 *static_cast<T*>(dest) = *value_;
92 return true;

96 // Set the result if not already set; returns true on first successful set.
97 bool set_value(T v) {
98 {
99 std::scoped_lock<std::mutex> lg(mutex_);
100 if (ready_) return false;
101 value_ = std::move(v);
102 ready_ = true;
103 }
104 cv_.notify_all();
105 return true;
106 }

108 // Consumer-side typed access (copy). Returns false if not ready.
109 bool get(T& out) const {
110 std::scoped_lock<std::mutex> lg(mutex_);
111 if (!ready_ || !value_.has_value()) return false;
112 out = *value_;
113 return true;
114 }

116private:
