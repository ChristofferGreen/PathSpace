199 // Serve typed keyboard events with peek/pop and optional blocking semantics.
200 // - If options.doPop is true: pop the front event into obj; otherwise peek without consuming.
201 // - If the queue is empty:
202 // * If options.doBlock is false: return NoObjectFound.
203 // * If options.doBlock is true: wait until timeout for an event to arrive; return Timeout on expiry.
204 auto out(Iterator const& /*path*/, InputMetadata const& inputMetadata, Out const& options, void* obj) -> std::optional<Error> override {
205 // Type-check: only support KeyboardEvent payloads here
206 if (inputMetadata.typeInfo != &typeid(Event)) {
207 return Error{Error::Code::InvalidType, "PathIOKeyboard only supports KeyboardEvent"};
208 }
209 if (obj == nullptr) {
210 return Error{Error::Code::MalformedInput, "Null output pointer for PathIOKeyboard::out"};
211 }
212
213 // Fast path: try without blocking
214 {
215 std::lock_guard<std::mutex> lg(mutex_);
216 if (!queue_.empty()) {
217 Event ev = queue_.front();
218 if (options.doPop) {
219 queue_.pop_front();
220 }
221 *reinterpret_cast<Event*>(obj) = ev;
222 return std::nullopt;
223 }
224 }
225
226 // No event and non-blocking read requested
227 if (!options.doBlock) {
228 return Error{Error::Code::NoObjectFound, "No keyboard event available"};
229 }
230
231 // Blocking path: wait until an event is available or timeout expires
232 auto deadline = std::chrono::steady_clock::now() +
233 std::chrono::duration_cast<std::chrono::steady_clock::duration>(options.timeout);
234
235 bool ready = waitFor(deadline, [&] { return !queue_.empty(); });
236 if (!ready) {
237 return Error{Error::Code::Timeout, "Timed out waiting for keyboard event"};
238 }
239
240 // Event is available now
241 {
242 std::lock_guard<std::mutex> lg(mutex_);
243 if (queue_.empty()) {
244 // Rare race: event consumed by another thread before we reacquired the lock
245 return Error{Error::Code::NoObjectFound, "No keyboard event available after wake"};
246 }
247 Event ev = queue_.front();
248 if (options.doPop) {
249 queue_.pop_front();
250 }
251 *reinterpret_cast<Event*>(obj) = ev;
252 }
253 return std::nullopt;
254 }

300 static CGEventRef KeyTapCallback(CGEventTapProxy /*proxy*/, CGEventType type, CGEventRef event, void* refcon) {
301 auto self = static_cast<PathIOKeyboard*>(refcon);
302 if (!self) return event;
303
304 // Re-enable tap if it gets disabled

308 }

310 switch (type) {
311 case kCGEventKeyDown: {

313 self->simulateKeyDown(keycode, /*modifiers=*/Mod_None, /*deviceId=*/0);
314
315 // Attempt to extract UTF-8 text for Text events
316 // Extract Unicode text from the key event (permission-less, local)
317 UniChar buffer[8] = {};
318 UniCharCount actual = 0;
319 CGEventKeyboardGetUnicodeString(event, (UniCharCount)(sizeof(buffer) / sizeof(buffer[0])), &actual, buffer);
320 if (actual > 0) {
321 CFStringRef s = CFStringCreateWithCharacters(kCFAllocatorDefault, buffer, actual);
322 if (s) {
323 CFIndex len = CFStringGetLength(s);
324 CFIndex maxSize = CFStringGetMaximumSizeForEncoding(len, kCFStringEncodingUTF8) + 1;
325 std::string utf8;
326 utf8.resize(static_cast<size_t>(maxSize), '\0');
327 CFIndex used = 0;
328 if (CFStringGetBytes(s, CFRangeMake(0, len), kCFStringEncodingUTF8, 0, false,
329 reinterpret_cast<UInt8*>(&utf8[0]), maxSize - 1, &used) && used > 0) {
330 utf8.resize(static_cast<size_t>(used));
331 self->simulateText(std::move(utf8), Mod_None, /*deviceId=*/0);
332 }
333 CFRelease(s);
334 }
335 }
336 break;
337 }
338 case kCGEventKeyUp: {
339 int keycode = static_cast<int>(CGEventGetIntegerValueField(event, kCGKeyboardEventKeycode));
340 self->simulateKeyUp(keycode, /*modifiers=*/Mod_None, /*deviceId=*/0);

349 void osInit_() {
350 CFRunLoopRef current = CFRunLoopGetCurrent();
351 if (runLoopRef_ && runLoopRef_ != current) {
352 osShutdown_();
353 }
354 if (eventTap_) {
355 // Already initialized for this runloop
356 return;
357 }
358
359 CGEventMask mask = CGEventMaskBit(kCGEventKeyDown) | CGEventMaskBit(kCGEventKeyUp);
360
361 eventTap_ = CGEventTapCreate(kCGHIDEventTap,
362 kCGHeadInsertEventTap,
363 kCGEventTapOptionDefault,
364 mask,
365 &KeyTapCallback,
366 this);
367 if (!eventTap_) {
368 // Could not create tap (permissions?). Fall back to simulation so example still functions.
369 fallbackActive_.store(true, std::memory_order_release);
370 return;
371 } else {
372 fallbackActive_.store(false, std::memory_order_release);
373 }
374
375 eventSrc_ = CFMachPortCreateRunLoopSource(kCFAllocatorDefault, eventTap_, 0);
376 runLoopRef_ = current;
377
378 if (eventSrc_ && runLoopRef_) {
379 CFRunLoopAddSource(runLoopRef_, eventSrc_, kCFRunLoopCommonModes);
380 CGEventTapEnable(eventTap_, true);
381 osReady_.store(true, std::memory_order_release);
382 }
383 }
384
385 void osShutdown_() {
386 osReady_.store(false, std::memory_order_release);
387 if (runLoopRef_ && eventSrc_) {
388 CFRunLoopRemoveSource(runLoopRef_, eventSrc_, kCFRunLoopCommonModes);
389 }
390 if (eventSrc_) {
391 CFRelease(eventSrc_);
392 eventSrc_ = nullptr;
393 }
394 if (eventTap_) {
395 CGEventTapEnable(eventTap_, false);
396 CFRelease(eventTap_);
397 eventTap_ = nullptr;
398 }
399 runLoopRef_ = nullptr;
400 // Disable fallback simulation on shutdown
401 fallbackActive_.store(false, std::memory_order_release);
402 }
