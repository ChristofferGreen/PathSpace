53 // Add or update a device under a class (e.g., "mouse", "keyboards").
54 // If the class name isn't one of the known tokens, it will be lowercased and used as-is.
55 void addSimulatedDevice(std::string cls, SimDevice dev) {
56 auto norm = normalizeClass_(cls);
57 {
58 std::lock_guard<std::mutex> lg(mutex_);
59 devices_[norm][dev.id] = std::move(dev);
60 }
61 notifyUpdates_(norm);
62 }

64 // Remove a device (no-op if not present).
65 void removeSimulatedDevice(std::string cls, int id) {
66 auto norm = normalizeClass_(cls);
67 bool changed = false;
68 {
69 std::lock_guard<std::mutex> lg(mutex_);
70 auto it = devices_.find(norm);
71 if (it != devices_.end()) {
72 auto& inner = it->second;
73 auto jt = inner.find(id);
74 if (jt != inner.end()) {
75 inner.erase(jt);
76 changed = true;
77 }
78 if (inner.empty()) {
79 devices_.erase(it);
80 }
81 }
82 }
83 if (changed) {
84 notifyUpdates_(norm);
85 }
86 }

104 // Serve discovery information as std::string at the paths described above.
105 auto out(Iterator const& path, InputMetadata const& inputMetadata, Out const& /*options*/, void* obj) -> std::optional<Error> override {
106 if (inputMetadata.typeInfo != &typeid(std::string)) {
107 return Error{Error::Code::TypeMismatch, "PathIODeviceDiscovery only supports std::string reads"};
108 }
109 if (obj == nullptr) {
110 return Error{Error::Code::MalformedInput, "Null output pointer for PathIODeviceDiscovery::out"};
111 }
112
113 // Use the iterator tail (current->end) to support nested mounts correctly.
114 std::string tail = std::string(path.currentToEnd()); // may be "" for root beneath mount
115 // Normalize: strip leading slash
116 std::string rel;
117 if (!tail.empty() && tail.front() == '/')
118 rel = tail.substr(1);
119 else
120 rel = tail;
121
122 // Split components
123 std::vector<std::string> parts;
124 {
125 size_t i = 0;
126 while (i < rel.size()) {
127 size_t j = rel.find('/', i);
128 if (j == std::string::npos) {
129 parts.emplace_back(rel.substr(i));
130 break;
131 } else {
132 parts.emplace_back(rel.substr(i, j - i));
133 i = j + 1;
134 }
135 }
136 // If rel is empty (root), parts stays empty
137 if (parts.size() == 1 && parts[0].empty()) parts.clear();
138 }
139
140 std::string outStr;
141 if (parts.empty()) {
142 // Root: list classes present
143 outStr = listClasses_();
144 } else if (parts.size() == 1) {
145 // Class: list device IDs
146 outStr = listDeviceIds_(parts[0]);
147 if (outStr.empty()) {
148 // If mounted under a parent space (mountPrefix_ set), treat single-component path
149 // as a request for the root class listing (useful for outMinimal under nested spaces).
150 std::string mountCopy;
151 {
152 std::lock_guard<std::mutex> lg(mutex_);
153 mountCopy = mountPrefix_;
154 }
155 if (!mountCopy.empty()) {
156 outStr = listClasses_();
157 } else {
158 return Error{Error::Code::NotFound, "No devices found for class: " + parts[0]};
159 }
160 }
161 } else if (parts.size() == 3) {
162 // Class / id / {meta|capabilities}
163 auto const& cls = parts[0];
164 auto const& idStr = parts[1];
165 auto const& leaf = parts[2];
166
167 int id = -1;
168 try {
169 id = std::stoi(idStr);
170 } catch (...) {
171 return Error{Error::Code::InvalidPath, "Invalid device id"};
172 }
173
174 if (leaf == "meta") {
175 auto metaOpt = getMeta_(cls, id);
176 if (!metaOpt.has_value())
177 return Error{Error::Code::NotFound, "Device not found"};
178 outStr = formatMeta_(*metaOpt);
179 } else if (leaf == "capabilities") {
180 auto capsOpt = getCapabilities_(cls, id);
181 if (!capsOpt.has_value())
182 return Error{Error::Code::NotFound, "Device not found"};
183 outStr = joinLines_(*capsOpt);
184 } else {
185 return Error{Error::Code::NotFound, "Unknown leaf under device: " + leaf};
186 }
187 } else {
188 return Error{Error::Code::InvalidPath, "Unsupported discovery path"};
189 }
190
191 *reinterpret_cast<std::string*>(obj) = std::move(outStr);
192 return std::nullopt;
193 }

217 static std::string normalizeClass_(std::string cls) {
218 auto s = toLower_(std::move(cls));
219 // Normalize common synonyms (canonicalize to singular 'mouse')
220 if (s == "mice") s = "mouse";
221 if (s == "keyboard") s = "keyboards";

227 std::string listClasses_() const {
228 std::set<std::string> classes;
229 std::lock_guard<std::mutex> lg(mutex_);
230 for (auto const& kv : devices_) {
231 if (!kv.second.empty())
232 classes.insert(kv.first);
233 }
234 std::ostringstream oss;
235 bool first = true;
236 for (auto const& c : classes) {
237 if (!first) oss << '\n';
238 oss << c;
239 first = false;
240 }
241 return oss.str();
242 }

244 std::string listDeviceIds_(std::string cls) const {
245 auto norm = normalizeClass_(std::move(cls));
246 std::ostringstream oss;
247 bool first = true;
248 std::lock_guard<std::mutex> lg(mutex_);
249 auto it = devices_.find(norm);
250 if (it == devices_.end()) return {};
251 for (auto const& kv : it->second) {
252 if (!first) oss << '\n';
253 oss << kv.first;
254 first = false;
255 }
256 return oss.str();

299 void notifyUpdates_(std::string const& cls) {
300 if (auto ctx = this->getContext()) {
301 std::string mount;
302 {
303 std::lock_guard<std::mutex> lg(mutex_);
304 mount = mountPrefix_;
305 }
306 if (!mount.empty()) {
307 ctx->notify(mount);
308 if (!cls.empty()) {
309 // Notify class subpath under the mount (best-effort)
310 ctx->notify(mount + "/" + cls);
311 }
312 } else {
313 ctx->notifyAll();
314 }
315 }
