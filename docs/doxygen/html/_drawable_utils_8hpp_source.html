63inline std::vector<std::uint32_t> build_draw_order(Scene::DrawableBucketSnapshot const& bucket) {
64 std::vector<std::uint32_t> order;
65 order.reserve(bucket.drawable_ids.size());
66 if (!bucket.opaque_indices.empty() || !bucket.alpha_indices.empty()) {
67 order.insert(order.end(), bucket.opaque_indices.begin(), bucket.opaque_indices.end());
68 order.insert(order.end(), bucket.alpha_indices.begin(), bucket.alpha_indices.end());
69 }
70 if (order.empty()) {
71 std::vector<std::uint32_t> opaque;
72 std::vector<std::uint32_t> alpha;
73 opaque.reserve(bucket.drawable_ids.size());
74 alpha.reserve(bucket.drawable_ids.size());
75 for (std::uint32_t i = 0; i < bucket.drawable_ids.size(); ++i) {
76 std::uint32_t flags = 0;
77 if (i < bucket.pipeline_flags.size()) {
78 flags = bucket.pipeline_flags[i];
79 }
80 if (PipelineFlags::is_alpha_pass(flags)) {
81 alpha.push_back(i);
82 } else {
83 opaque.push_back(i);
84 }
85 }
86 if (opaque.empty() && alpha.empty()) {
87 opaque.resize(bucket.drawable_ids.size());
88 std::iota(opaque.begin(), opaque.end(), 0u);
89 }
90 order.insert(order.end(), opaque.begin(), opaque.end());
91 order.insert(order.end(), alpha.begin(), alpha.end());
92 }
93 return order;
94}

96inline bool point_inside_clip(float x,
97 float y,
98 Scene::DrawableBucketSnapshot const& bucket,
99 std::size_t drawable_index) {
100 if (drawable_index >= bucket.clip_head_indices.size()) {
101 return true;
102 }
103 auto node_index = bucket.clip_head_indices[drawable_index];
104 if (node_index < 0) {
105 return true;
106 }
107 while (node_index >= 0) {
108 if (static_cast<std::size_t>(node_index) >= bucket.clip_nodes.size()) {
109 break;
110 }
111 auto const& node = bucket.clip_nodes[static_cast<std::size_t>(node_index)];
112 if (node.type == Scene::ClipNodeType::Rect) {
113 auto const& rect = node.rect;
114 if (x < rect.min_x || x > rect.max_x || y < rect.min_y || y > rect.max_y) {
115 return false;
116 }
117 }
118 node_index = node.next;
119 }
120 return true;
121}
